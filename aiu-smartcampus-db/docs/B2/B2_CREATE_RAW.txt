CREATE TABLE aiu_timeseries.sensor_readings_raw\n(\n    `sensor_id` UInt32,\n    `room_id` UInt32,\n    `sensor_type` LowCardinality(String),\n    `ts` DateTime64(3, \'UTC\'),\n    `value` Float64,\n    `status` LowCardinality(String)\n)\nENGINE = MergeTree\nPARTITION BY toDate(ts)\nORDER BY (sensor_id, ts)\nTTL toDateTime(ts) + toIntervalDay(365)\nSETTINGS index_granularity = 8192
